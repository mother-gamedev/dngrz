# DNGRZ v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a branded, locally-playable two-player + AI pitch/bat prototype that answers the foundational design question — *is the pitch/bat duel fun on its own?*

**Architecture:** Manual trajectory calculation for ball physics (deterministic, not RigidBody3D). CSG primitives for greybox field. Pure GDScript classes for logic (testable without scenes). GdUnit4 for testing. State machine pattern for game flow. **Godot Theme + design tokens applied from day one** so all UI matches the design handoff visually. AI controllers as drop-in replacements for human controllers, enabling solo feel-testing.

**Tech Stack:** Godot 4.5, GDScript, GdUnit4, CSG nodes, Forward Plus renderer, Godot Theme resources, Inter Tight + Archivo Black + JetBrains Mono fonts (Google Fonts)

- [ ] **Step 6: Build chamfered button helper**

Create `themes/styles/chamfered_button.gd`:

```gdscript
class_name ChamferedButton extends Button

@export var notch_px: float = 10.0
@export var fill_color: Color = Color("#FFCC00")
@export var hover_color: Color = Color("#FFB400")

var _polygon: Polygon2D
var _hovered := false

func _ready() -> void:
    flat = true
    _polygon = Polygon2D.new()
    add_child(_polygon)
    move_child(_polygon, 0)
    mouse_entered.connect(_on_hover_changed.bind(true))
    mouse_exited.connect(_on_hover_changed.bind(false))
    resized.connect(_update_polygon)
    _update_polygon()

func _on_hover_changed(h: bool) -> void:
    _hovered = h
    _update_polygon()

func _update_polygon() -> void:
    if _polygon == null: return
    var w := size.x
    var h := size.y
    var n := notch_px
    _polygon.polygon = PackedVector2Array([
        Vector2(0, 0),
        Vector2(w - n, 0),
        Vector2(w, n),
        Vector2(w, h),
        Vector2(0, h),
    ])
    _polygon.color = hover_color if _hovered else fill_color
```

class_name PitcherAI extends Node

class Decision:
    var pitch_type: PitchTypes.Type
    var target: Vector3
    var accuracy: float

    func _init(pt: PitchTypes.Type, t: Vector3, a: float) -> void:
        pitch_type = pt
        target = t
        accuracy = a

@export var pitcher_controller: NodePath
@export var auto_pitch_interval: float = 2.5
@export var enabled: bool = false

var _time_since_last := 0.0
var _history: Array = []

func _process(delta: float) -> void:
    if not enabled: return
    _time_since_last += delta
    if _time_since_last >= auto_pitch_interval:
        _time_since_last = 0.0
        _throw()

func _throw() -> void:
    if pitcher_controller.is_empty(): return
    var ctrl := get_node_or_null(pitcher_controller)
    if ctrl == null: return
    var d := decide(0, 0, _history)
    ctrl.request_pitch(d.pitch_type, d.target, d.accuracy)
    _history.append(d.pitch_type)
    if _history.size() > 5: _history.pop_front()

# Pure decision function — testable without scene tree.
func decide(balls: int, strikes: int, history: Array) -> Decision:
    var pitch_type := _select_pitch_type(balls, strikes, history)
    var target := _select_target(balls, strikes, pitch_type)
    var accuracy := _accuracy_for(balls, strikes)
    return Decision.new(pitch_type, target, accuracy)

func _select_pitch_type(balls: int, strikes: int, history: Array) -> PitchTypes.Type:
    var candidates: Array = [
        PitchTypes.Type.FASTBALL,
        PitchTypes.Type.CURVEBALL,
        PitchTypes.Type.SLIDER,
        PitchTypes.Type.CHANGEUP,
    ]
    if history.size() > 0 and randf() < 0.7:
        candidates.erase(history[history.size() - 1])

    if balls >= 2 and strikes == 0 and randf() < 0.7:
        return PitchTypes.Type.FASTBALL

    if strikes == 2 and balls <= 1 and randf() < 0.6:
        if candidates.size() > 0:
            return candidates[randi() % candidates.size()]
        return PitchTypes.Type.SLIDER

    if candidates.size() == 0:
        return PitchTypes.Type.FASTBALL
    return candidates[randi() % candidates.size()]

func _select_target(balls: int, strikes: int, pitch_type: PitchTypes.Type) -> Vector3:
    if balls >= 3 and strikes == 0:
        return Vector3(0.0, 0.8, 0.0)
    if strikes == 2 and balls <= 1:
        return Vector3(randf_range(-0.3, 0.3), 0.3, 0.0)
    return Vector3(randf_range(-0.25, 0.25), randf_range(0.55, 1.05), 0.0)

func _accuracy_for(balls: int, strikes: int) -> float:
    if balls == 3: return 0.95
    if strikes == 2: return 0.90
    return 0.80

```gdscript
class_name BatterAI extends Node

class Decision:
    var swing: bool
    var timing_offset: float
    var placement: Vector2

    func _init(s: bool, t: float, p: Vector2) -> void:
        swing = s
        timing_offset = t
        placement = p

@export var batter_controller: NodePath
@export var enabled: bool = false
@export var skill: float = 0.7  # 0..1

# Pure decision function. `predicted_ball_at_plate` is the ball's predicted
# position when it crosses the plate plane.
func decide(predicted_ball_at_plate: Vector3, balls: int, strikes: int) -> Decision:
    var in_zone := StrikeZone.is_strike(predicted_ball_at_plate)
    var should := _should_swing(predicted_ball_at_plate, in_zone, balls, strikes)
    if not should:
        return Decision.new(false, 0.0, Vector2.ZERO)
    var timing := _swing_timing()
    var placement := _swing_placement()
    return Decision.new(true, timing, placement)

func _should_swing(ball: Vector3, in_zone: bool, balls: int, strikes: int) -> bool:
    if in_zone:
        return randf() < 0.85
    var d := _distance_outside_zone(ball)
    if strikes == 2 and d < 0.15:
        return randf() < 0.75
    if balls >= 3:
        return randf() < 0.05
    if d < 0.08:
        return randf() < 0.4
    return randf() < 0.05

func _distance_outside_zone(ball: Vector3) -> float:
    var half_w := FieldConstants.STRIKE_ZONE_WIDTH / 2.0
    var dx := maxf(0.0, absf(ball.x) - half_w)
    var dy := 0.0
    if ball.y < FieldConstants.STRIKE_ZONE_BOTTOM:
        dy = FieldConstants.STRIKE_ZONE_BOTTOM - ball.y
    elif ball.y > FieldConstants.STRIKE_ZONE_TOP:
        dy = ball.y - FieldConstants.STRIKE_ZONE_TOP
    return Vector2(dx, dy).length()

func _swing_timing() -> float:
    var spread := lerpf(0.08, 0.015, skill)
    return randf_range(-spread, spread)

func _swing_placement() -> Vector2:
    var noise_scale := lerpf(0.1, 0.02, skill)
    return Vector2(
        randf_range(-noise_scale, noise_scale),
        randf_range(-0.02, noise_scale * 0.5)
    )
```

**Files:**

**Reference:** Design handoff screen #11.

- [ ] **Step 1: Build the scene**

In editor:
- Root: Control, full_rect
- `ResultBand` (PanelContainer) — top 25%, gradient background (stub: linear-gradient between two arbitrary faction colors). Contains:
  - Label "Outcome" (Archivo Black 96pt) — set to "WIN" or "LOSS"
  - Label "Score" (Archivo Black 36pt) — "5-3 · 5 INNINGS"
- `LineScore` (GridContainer, 9 columns) — fills middle 40%. Headers: TEAM 1 2 3 4 5 R H E. Rows added at runtime.
- `SummaryCard` (Panel) — single-line summary; v1 shows "+0 HYPE (POST-V1)".
- `ButtonStack` (VBox, bottom right) — 3 ChamferedButton instances: "Rematch", "Next", "Leave".

- [ ] **Step 2: Build the script**

```gdscript
class_name PostMatch extends Control

signal rematch_pressed
signal next_pressed
signal leave_pressed

func _ready() -> void:
    ($ButtonStack/Rematch as Button).pressed.connect(func(): rematch_pressed.emit())
    ($ButtonStack/Next as Button).pressed.connect(func(): next_pressed.emit())
    ($ButtonStack/Leave as Button).pressed.connect(func(): leave_pressed.emit())

func display(home_score: int, away_score: int, line_score: Array, hits: Dictionary, errors: Dictionary) -> void:
    var home_won := home_score > away_score
    ($ResultBand/Outcome as Label).text = "WIN" if home_won else "LOSS"
    ($ResultBand/Score as Label).text = "%d-%d · 5 INNINGS" % [max(home_score, away_score), min(home_score, away_score)]

    var grid := $LineScore as GridContainer
    for child in grid.get_children(): child.queue_free()
    _add_header(grid)
    _add_team_row(grid, "AWAY", line_score[0], away_score, hits.get("away", 0), errors.get("away", 0))
    _add_team_row(grid, "HOME", line_score[1], home_score, hits.get("home", 0), errors.get("home", 0))

func _add_header(grid: GridContainer) -> void:
    for text in ["TEAM", "1", "2", "3", "4", "5", "R", "H", "E"]:
        var lbl := Label.new()
        lbl.text = text
        lbl.add_theme_color_override("font_color", Colors.TEXT_DIM)
        grid.add_child(lbl)

func _add_team_row(grid: GridContainer, team: String, innings: Array, r: int, h: int, e: int) -> void:
    var team_lbl := Label.new()
    team_lbl.text = team
    grid.add_child(team_lbl)
    for inning_runs in innings:
        var l := Label.new()
        l.text = str(inning_runs)
        grid.add_child(l)
    for val in [r, h, e]:
        var l := Label.new()
        l.text = str(val)
        l.add_theme_color_override("font_color", Colors.BRAND)
        grid.add_child(l)
```
